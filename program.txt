# Проверка числа N на простоту (делением на последовательность делителей)
# Вход: R7 - исходное число A
# Выход: R6 - 1 -> число простое, иначе - минимальный делитель
# Испорчены регистры: R0 R1 R2 R5 R6

MOV R7 13       # исходное число A

# отдельная проверка на четность
MOV R5 2
MOD R2 R7 R5
JZ EXIT

# делим интервал на 2
DIV R0 R7 R5

# цикл i=3, i<=A//2, i++
MOV R6 1
ADD R1 R5 R6
DO:
    SUB R2 R0 R1       # i<=A//2?
    JZ EXIT

    MOD R2 R7 R1    # i=делитель?
    JZ EXIT 

    ADD R1 R1 R5    # i++
    JMP DO

EXIT:
    MOV R5 0
    CMP R2 R5   # найден делитель? 
    JNZ SKIP

    MOV R6 R1   # да - делитель в результат

SKIP:           # нет - число простое
    HLT
