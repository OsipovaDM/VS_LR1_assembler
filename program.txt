# В одной строке читается только одна операция/метка. Остальной ткест после воспринимается списком операндов/игнорируется

# Числа (только положительные)
MOV R0 42        # OK
MOV R1 0         # OK
MOV R2 65535     # OK (максимальное)

# Регистры
MOV R5 R0        # OK
JMP R8           # OK: R8 - метка
R8: NOP

# Метки
JMP START        # OK
START: NOP
# Тест всех режимов MOV
GO:
MOV R0 42        # регистр <- число
MOV R1 100       # регистр <- число
MOV [30] R0      # память <- регистр
MOV R2 [30]      # регистр <- память
MOV R3 R1        # регистр <- регистр
HLT
