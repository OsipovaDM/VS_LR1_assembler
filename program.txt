# Программа вычисления НОД (алгоритм Евклида)
# Входные данные: числа в R0 и R1
# Результат: НОД в R0

# Пример: R0 = 12 R1 = 18 -> результат: 6



# Третий вариант (наиболее эффективный - с использованием остатка)
# Использует MOD для быстрого вычисления
GCD_WITH_MOD:
    MOV R0 113        # Загружаем первое число a (12)
    MOV R1 99         # Загружаем второе число b (8)
    MOV R7 0
    
DO:
    CMP R1 R7       # B = 0?
    JZ EXIT
    CMP R0 R7       # A = 0?
    JZ EXIT

    # Нет
    SUB R3 R0 R1    # A < B?
    JZ BhA

    # Нет
    MOD R0 R0 R1    # A = A % B
    JMP DO

BhA:
    # Да
    MOD R1 R1 R0    # B = B % A
    JMP DO

EXIT:
    # Да
    # Результат в R0
    ADD R0 R0 R1    # A = A + B
    HLT
