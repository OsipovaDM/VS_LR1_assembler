# Определение минимального элемента в массиве фиксированной длины
# Все эл-ты массива положительные?
# Первый эл-т располагается в памяти по адресу [0]
# Вход:     R7 - кол-во эл-тов массива
# Выход:    R6 - значение мин. эл-та

START:

    MOV R6 65535    # result = 2^16 - 1
    MOV R0 0    # i = 0
    MOV R3 1

    # while i < N, i++
DO:

    SUB R2 R7 R0   # N >= i ?
    JZ EXIT

    MOV R1 [R0] # Нет. Чтение i эл-та
    SUB R2 R6 R1   # result >= эл-та
    JZ SKIP

    MOV R6 R1   # Нет. result = эл-т
SKIP:           # Да.  result = result
    
    ADD R0 R0 R3
    JMP DO
EXIT:           # Да. Выход за пределы массива. Проверка завершена
